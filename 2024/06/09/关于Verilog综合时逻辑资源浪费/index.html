            CNT_RISE_MAX    =   28'd12_500_000  ;   //è½¯ä»¶é—¸é—¨æ‹‰é«˜è®¡æ•°å€¼

    else    if(gate_a == 1'b0)
        cnt_clk_test    <=  48'd0;
    else    if(gate_a == 1'b1)
        cnt_clk_test    <=  cnt_clk_test + 1'b1;

//gate_a_stand:å®é™…é—¸é—¨æ‰“ä¸€æ‹(æ ‡å‡†æ—¶é’Ÿä¸‹)

//gate_a_test:å®é™…é—¸é—¨æ‰“ä¸€æ‹(å¾…æ£€æµ‹æ—¶é’Ÿä¸‹)
